    58‚Üí    58‚Üíclass ScaleUpSolver(Solver):
    59‚Üí    59‚Üí    """Output is input scaled by an integer factor k (k=2,3,4)."""
    60‚Üí    60‚Üí    name = "scale_up"
    61‚Üí    61‚Üí
    62‚Üí    62‚Üí    def attempt(self, task: dict) -> Optional[onnx.ModelProto]:
    63‚Üí    63‚Üí        pairs = get_pairs(task)
    64‚Üí    64‚Üí        k = None
    65‚Üí    65‚Üí        for inp, out in pairs:
    66‚Üí    66‚Üí            if inp.shape[0] == 0 or inp.shape[1] == 0:
    67‚Üí    67‚Üí                return None
    68‚Üí    68‚Üí            kh = out.shape[0] / inp.shape[0]
    69‚Üí    69‚Üí            kw = out.shape[1] / inp.shape[1]
    70‚Üí    70‚Üí            if not kh.is_integer() or not kw.is_integer() or kh != kw:
    71‚Üí    71‚Üí                return None
    72‚Üí    72‚Üí            kk = int(kh)
    73‚Üí    73‚Üí            if k is None:
    74‚Üí    74‚Üí                k = kk
    75‚Üí    75‚Üí            elif k != kk:
    76‚Üí    76‚Üí                return None
    77‚Üí    77‚Üí        if k is None or k <= 1:
    78‚Üí    78‚Üí            return None
    79‚Üí    79‚Üí        for inp, out in pairs:
    80‚Üí    80‚Üí            scaled = np.repeat(np.repeat(inp, k, axis=0), k, axis=1)
    81‚Üí    81‚Üí            if scaled.shape != out.shape or not np.array_equal(scaled, out):
    82‚Üí    82‚Üí                return None
    83‚Üí    83‚Üí        nodes = [
    84‚Üí    84‚Üí            h.make_node("Constant", [], ["roi"],
    85‚Üí    85‚Üí                         value=h.make_tensor("roi_v", TensorProto.FLOAT, [0], [])),
    86‚Üí    86‚Üí            h.make_node("Constant", [], ["scales"],
    87‚Üí    87‚Üí                         value=h.make_tensor("scales_v", TensorProto.FLOAT, [4],
    88‚Üí    88‚Üí                                              [1.0, 1.0, float(k), float(k)])),
    89‚Üí    89‚Üí            h.make_node("Resize", [INPUT_NAME, "roi", "scales"], [OUTPUT_NAME],
    90‚Üí    90‚Üí                         mode="nearest",
    91‚Üí    91‚Üí                         nearest_mode="floor",
    92‚Üí    92‚Üí                         coordinate_transformation_mode="asymmetric"),
    93‚Üí    93‚Üí        ]
    94‚Üí    94‚Üí        return _make_simple_model(nodes, name="scale_up")

   123‚Üí   123‚Üíclass ShiftSolver(Solver):
   124‚Üí   124‚Üí    """Output = input shifted by (dh, dw). Pads with 0 (color 0).
   125‚Üí   125‚Üí
   126‚Üí   126‚Üí    Implemented using Pad + Slice (or just Slice with appropriate offsets).
   127‚Üí   127‚Üí    Output dimensions = input dimensions (validator crops).
   128‚Üí   128‚Üí    """
   129‚Üí   129‚Üí    name = "shift"
   130‚Üí   130‚Üí
   131‚Üí   131‚Üí    def attempt(self, task: dict) -> Optional[onnx.ModelProto]:
   132‚Üí   132‚Üí        pairs = get_pairs(task)
   133‚Üí   133‚Üí        # Determine the shift (dh, dw) ‚Äî must be consistent across all pairs
   134‚Üí   134‚Üí        shift = None
   135‚Üí   135‚Üí        for inp, out in pairs:
   136‚Üí   136‚Üí            if inp.shape != out.shape:
   137‚Üí   137‚Üí                return None
   138‚Üí   138‚Üí            # Find the shift by trying all possibilities
   139‚Üí   139‚Üí            H, W = inp.shape
   140‚Üí   140‚Üí            found = None
   141‚Üí   141‚Üí            for dh in range(-H + 1, H):
   142‚Üí   142‚Üí                for dw in range(-W + 1, W):
   143‚Üí   143‚Üí                    # Apply shift: out[i, j] = inp[i - dh, j - dw] if in bounds else 0
   144‚Üí   144‚Üí                    candidate = np.zeros_like(inp)
   145‚Üí   145‚Üí                    for i in range(H):
   146‚Üí   146‚Üí                        for j in range(W):
   147‚Üí   147‚Üí                            si = i - dh
   148‚Üí   148‚Üí                            sj = j - dw
   149‚Üí   149‚Üí                            if 0 <= si < H and 0 <= sj < W:
   150‚Üí   150‚Üí                                candidate[i, j] = inp[si, sj]
   151‚Üí   151‚Üí                    if np.array_equal(candidate, out):
   152‚Üí   152‚Üí                        found = (dh, dw)
   153‚Üí   153‚Üí                        break
   154‚Üí   154‚Üí                if found:
   155‚Üí   155‚Üí                    break
   156‚Üí   156‚Üí            if found is None:
   157‚Üí   157‚Üí                return None
   158‚Üí   158‚Üí            if shift is None:
   159‚Üí   159‚Üí                shift = found
   160‚Üí   160‚Üí            elif shift != found:
   161‚Üí   161‚Üí                return None
   162‚Üí   162‚Üí        if shift is None:
   163‚Üí   163‚Üí            return None
   164‚Üí   164‚Üí        dh, dw = shift
   165‚Üí   165‚Üí        if dh == 0 and dw == 0:
   166‚Üí   166‚Üí            return dsl.identity()
   167‚Üí   167‚Üí        return _shift_model(dh, dw)
   168‚Üí   168‚Üí
   169‚Üí   169‚Üí
   170‚Üí   170‚Üídef _shift_model(dh: int, dw: int) -> onnx.ModelProto:
   171‚Üí   171‚Üí    """Build a network that shifts the input by (dh, dw) ‚Äî out[i,j] = inp[i-dh, j-dw].
   172‚Üí   172‚Üí
   173‚Üí   173‚Üí    Uses Slice with negative-start trickery inside the 30x30 frame, then
   174‚Üí   174‚Üí    places the result back at the top-left.
   175‚Üí   175‚Üí    """
   176‚Üí   176‚Üí    # Strategy: Slice the input from (dh, dw) to (dh+30, dw+30) within the 30x30 frame.
   177‚Üí   177‚Üí    # If dh > 0: source starts at row dh, output starts at row 0.
   178‚Üí   178‚Üí    # If dh < 0: source starts at row 0, output starts at row -dh (so we Pad top by -dh).
   179‚Üí   179‚Üí    # Simpler: Pad input with zeros on the side opposite to the shift direction,
   180‚Üí   180‚Üí    # then slice the top-left 30x30.
   181‚Üí   181‚Üí    #
   182‚Üí   182‚Üí    # Equivalent: use Slice with starts=[max(0,dh), max(0,dw)] ends=[30+max(0,dh), 30+max(0,dw)]
   183‚Üí   183‚Üí    # But this loses cells that fall off the bottom.
   184‚Üí   184‚Üí    # Better: Pad then Slice.
   185‚Üí   185‚Üí    #
   186‚Üí   186‚Üí    # Pad: top = max(0, -dh), bottom = max(0, dh), left = max(0, -dw), right = max(0, dw)
   187‚Üí   187‚Üí    # Then slice top-left 30x30.
   188‚Üí   188‚Üí
   189‚Üí   189‚Üí    top = max(0, -dh)
   190‚Üí   190‚Üí    bottom = max(0, dh)
   191‚Üí   191‚Üí    left = max(0, -dw)
   192‚Üí   192‚Üí    right = max(0, dw)
   193‚Üí   193‚Üí    pads = [0, 0, top, left, 0, 0, bottom, right]
   194‚Üí   194‚Üí    nodes = [
   195‚Üí   195‚Üí        h.make_node("Constant", [], ["pads_c"],
   196‚Üí   196‚Üí                     value=h.make_tensor("pads_v", TensorProto.INT64, [8], pads)),
   197‚Üí   197‚Üí        h.make_node("Constant", [], ["val_c"],
   198‚Üí   198‚Üí                     value=h.make_tensor("val_v", TensorProto.FLOAT, [], [0.0])),
   199‚Üí   199‚Üí        h.make_node("Pad", [INPUT_NAME, "pads_c", "val_c"], ["padded"],
   200‚Üí   200‚Üí                     mode="constant"),
   201‚Üí   201‚Üí        # Slice padded (30+|dh| x 30+|dw|) to top-left 30x30
   202‚Üí   202‚Üí        h.make_node("Constant", [], ["starts"],
   203‚Üí   203‚Üí                     value=h.make_tensor("starts_v", TensorProto.INT64, [4], [0, 0, 0, 0])),
   204‚Üí   204‚Üí        h.make_node("Constant", [], ["ends"],
   205‚Üí   205‚Üí                     value=h.make_tensor("ends_v", TensorProto.INT64, [4],
   206‚Üí   206‚Üí                                          [1, NUM_COLORS, MAX_GRID, MAX_GRID])),
   207‚Üí   207‚Üí        h.make_node("Constant", [], ["axes"],
   208‚Üí   208‚Üí                     value=h.make_tensor("axes_v", TensorProto.INT64, [4], [0, 1, 2, 3])),
   209‚Üí   209‚Üí        h.make_node("Slice", ["padded", "starts", "ends", "axes"], [OUTPUT_NAME]),
   210‚Üí   210‚Üí    ]
   211‚Üí   211‚Üí    return _make_simple_model(nodes, name=f"shift_{dh}_{dw}")
   212‚Üí   212‚Üí
   213‚Üí   213‚Üí

   219‚Üí   219‚Üíclass TileSolver(Solver):
   220‚Üí   220‚Üí    """Output = input pattern repeated (tile_factor_x, tile_factor_y) times.
   221‚Üí   221‚Üí
   222‚Üí   222‚Üí    E.g., 3x3 input tiled 3x3 -> 9x9 output where the 3x3 pattern repeats.
   223‚Üí   223‚Üí    Implemented using ONNX Tile op.
   224‚Üí   224‚Üí    """
   225‚Üí   225‚Üí    name = "tile"
   226‚Üí   226‚Üí
   227‚Üí   227‚Üí    def attempt(self, task: dict) -> Optional[onnx.ModelProto]:
   228‚Üí   228‚Üí        pairs = get_pairs(task)
   229‚Üí   229‚Üí        # Determine tile factor
   230‚Üí   230‚Üí        tf = None
   231‚Üí   231‚Üí        for inp, out in pairs:
   232‚Üí   232‚Üí            if inp.shape[0] == 0 or inp.shape[1] == 0:
   233‚Üí   233‚Üí                return None
   234‚Üí   234‚Üí            th = out.shape[0] / inp.shape[0]
   235‚Üí   235‚Üí            tw = out.shape[1] / inp.shape[1]
   236‚Üí   236‚Üí            if not th.is_integer() or not tw.is_integer():
   237‚Üí   237‚Üí                return None
   238‚Üí   238‚Üí            t = (int(th), int(tw))
   239‚Üí   239‚Üí            if tf is None:
   240‚Üí   240‚Üí                tf = t
   241‚Üí   241‚Üí            elif tf != t:
   242‚Üí   242‚Üí                return None
   243‚Üí   243‚Üí        if tf is None or tf == (1, 1):
   244‚Üí   244‚Üí            return None
   245‚Üí   245‚Üí        # Verify
   246‚Üí   246‚Üí        for inp, out in pairs:
   247‚Üí   247‚Üí            tiled = np.tile(inp, tf)
   248‚Üí   248‚Üí            if tiled.shape != out.shape or not np.array_equal(tiled, out):
   249‚Üí   249‚Üí                return None
   250‚Üí   250‚Üí        th, tw = tf
   251‚Üí   251‚Üí        # Build Tile model: repeats = [1, 1, th, tw]
   252‚Üí   252‚Üí        nodes = [
   253‚Üí   253‚Üí            h.make_node("Constant", [], ["repeats"],
   254‚Üí   254‚Üí                         value=h.make_tensor("repeats_v", TensorProto.INT64, [4],
   255‚Üí   255‚Üí                                              [1, 1, th, tw])),
   256‚Üí   256‚Üí            h.make_node("Tile", [INPUT_NAME, "repeats"], [OUTPUT_NAME]),
   257‚Üí   257‚Üí        ]
   258‚Üí   258‚Üí        return _make_simple_model(nodes, name=f"tile_{th}x{tw}")
   259‚Üí   259‚Üí
   260‚Üí   260‚Üí
   261‚Üí   261‚Üí# ---------------------------------------------------------------------------
   262‚Üí   262‚Üí# KroneckerSolver ‚Äî out[i*k+a, j*k+b] = in[a,b] if in[i,j]!=0 else 0
   263‚Üí   263‚Üí# ---------------------------------------------------------------------------

   266‚Üí   266‚Üíclass KroneckerSolver(Solver):
   267‚Üí   267‚Üí    """Kronecker-like product: each non-zero input cell becomes a copy of the
   268‚Üí   268‚Üí    input pattern in the output; zero input cells become zero blocks.
   269‚Üí   269‚Üí
   270‚Üí   270‚Üí    Matches tasks like 007bbfb7 (3x3 -> 9x9 conditional tiling).
   271‚Üí   271‚Üí
   272‚Üí   272‚Üí    Implementation: out = Tile(input, kxk) * Resize(input, kxk) where
   273‚Üí   273‚Üí    Resize gives the "block value" and Tile gives the "within-block value".
   274‚Üí   274‚Üí    But we need to handle the "is block non-zero" check correctly in one-hot.
   275‚Üí   275‚Üí    """
   276‚Üí   276‚Üí    name = "kronecker"
   277‚Üí   277‚Üí
   278‚Üí   278‚Üí    def attempt(self, task: dict) -> Optional[onnx.ModelProto]:
   279‚Üí   279‚Üí        pairs = get_pairs(task)
   280‚Üí   280‚Üí        # Determine k
   281‚Üí   281‚Üí        k = None
   282‚Üí   282‚Üí        for inp, out in pairs:
   283‚Üí   283‚Üí            if inp.shape[0] == 0 or inp.shape[1] == 0:
   284‚Üí   284‚Üí                return None
   285‚Üí   285‚Üí            if inp.shape[0] * inp.shape[1] == 0:
   286‚Üí   286‚Üí                return None
   287‚Üí   287‚Üí            # Kronecker requires input to be kxk (square)
   288‚Üí   288‚Üí            if inp.shape[0] != inp.shape[1]:
   289‚Üí   289‚Üí                return None
   290‚Üí   290‚Üí            kh = out.shape[0] / inp.shape[0]
   291‚Üí   291‚Üí            kw = out.shape[1] / inp.shape[1]
   292‚Üí   292‚Üí            if not kh.is_integer() or not kw.is_integer() or kh != kw:
   293‚Üí   293‚Üí                return None
   294‚Üí   294‚Üí            kk = int(kh)
   295‚Üí   295‚Üí            # Input size must equal k (Kronecker pattern: input is the "tile" itself)
   296‚Üí   296‚Üí            if inp.shape[0] != kk:
   297‚Üí   297‚Üí                return None
   298‚Üí   298‚Üí            if k is None:
   299‚Üí   299‚Üí                k = kk
   300‚Üí   300‚Üí            elif k != kk:
   301‚Üí   301‚Üí                return None
   302‚Üí   302‚Üí        if k is None or k <= 1:
   303‚Üí   303‚Üí            return None
   304‚Üí   304‚Üí        # Verify: out[i*k+a, j*k+b] = in[a, b] if in[i, j] != 0 else 0
   305‚Üí   305‚Üí        for inp, out in pairs:
   306‚Üí   306‚Üí            H, W = inp.shape
   307‚Üí   307‚Üí            for i in range(H):
   308‚Üí   308‚Üí                for j in range(W):
   309‚Üí   309‚Üí                    block = out[i*k:(i+1)*k, j*k:(j+1)*k]
   310‚Üí   310‚Üí                    if inp[i, j] == 0:
   311‚Üí   311‚Üí                        if not np.array_equal(block, np.zeros((k, k))):
   312‚Üí   312‚Üí                            return None
   313‚Üí   313‚Üí                            break
   314‚Üí   314‚Üí                    else:
   315‚Üí   315‚Üí                        if not np.array_equal(block, inp):
   316‚Üí   316‚Üí                            return None
   317‚Üí   317‚Üí                            break
   318‚Üí   318‚Üí        return _kronecker_model(k)

   320‚Üí   320‚Üí
   321‚Üí   321‚Üídef _kronecker_model(k: int) -> onnx.ModelProto:
   322‚Üí   322‚Üí    """Build a Kronecker-product network with factor k.
   323‚Üí   323‚Üí
   324‚Üí   324‚Üí    Requires input to be kxk (placed in top-left of 30x30 frame, rest zero).
   325‚Üí   325‚Üí
   326‚Üí   326‚Üí    For one-hot input I (1, 10, 30, 30):
   327‚Üí   327‚Üí      S = Slice(I, top-left kxk)              ‚Üí (1, 10, k, k)
   328‚Üí   328‚Üí      T1 = Resize(S, scale=k)                 ‚Üí (1, 10, k*k, k*k)  block-constant tiling

   330‚Üí   330‚Üí      gate = 1 - T1[0:1]                      ‚Üí 1 where block non-zero, else 0
   331‚Üí   331‚Üí      O_small = T2 * gate                     ‚Üí (1, 10, k*k, k*k)
   332‚Üí   332‚Üí      O = Pad(O_small, to 30x30)              ‚Üí (1, 10, 30, 30)

   334‚Üí   334‚Üí    kk = k * k
   335‚Üí   335‚Üí    pad_h = MAX_GRID - kk
   336‚Üí   336‚Üí    pad_w = MAX_GRID - kk
   337‚Üí   337‚Üí    nodes = [
   338‚Üí   338‚Üí        # Slice input to (1, 10, k, k) ‚Äî top-left kxk
   339‚Üí   339‚Üí        h.make_node("Constant", [], ["s_starts"],
   340‚Üí   340‚Üí                     value=h.make_tensor("s_starts_v", TensorProto.INT64, [4], [0, 0, 0, 0])),
   341‚Üí   341‚Üí        h.make_node("Constant", [], ["s_ends"],
   342‚Üí   342‚Üí                     value=h.make_tensor("s_ends_v", TensorProto.INT64, [4], [1, NUM_COLORS, k, k])),
   343‚Üí   343‚Üí        h.make_node("Constant", [], ["s_axes"],
   344‚Üí   344‚Üí                     value=h.make_tensor("s_axes_v", TensorProto.INT64, [4], [0, 1, 2, 3])),
   345‚Üí   345‚Üí        h.make_node("Slice", [INPUT_NAME, "s_starts", "s_ends", "s_axes"], ["S"]),
   346‚Üí   346‚Üí        # T1 = Resize(S, scale=k) ‚Üí (1, 10, k*k, k*k)
   347‚Üí   347‚Üí        h.make_node("Constant", [], ["roi1"],
   348‚Üí   348‚Üí                     value=h.make_tensor("roi1_v", TensorProto.FLOAT, [0], [])),
   349‚Üí   349‚Üí        h.make_node("Constant", [], ["scales1"],
   350‚Üí   350‚Üí                     value=h.make_tensor("scales1_v", TensorProto.FLOAT, [4],
   351‚Üí   351‚Üí                                          [1.0, 1.0, float(k), float(k)])),
   352‚Üí   352‚Üí        h.make_node("Resize", ["S", "roi1", "scales1"], ["T1"],
   353‚Üí   353‚Üí                     mode="nearest", nearest_mode="floor",
   354‚Üí   354‚Üí                     coordinate_transformation_mode="asymmetric"),
   355‚Üí   355‚Üí        # T2 = Tile(S, repeats=[1,1,k,k]) ‚Üí (1, 10, k*k, k*k)
   356‚Üí   356‚Üí        h.make_node("Constant", [], ["repeats2"],
   357‚Üí   357‚Üí                     value=h.make_tensor("repeats2_v", TensorProto.INT64, [4],
   358‚Üí   358‚Üí                                          [1, 1, k, k])),
   359‚Üí   359‚Üí        h.make_node("Tile", ["S", "repeats2"], ["T2"]),

   361‚Üí   361‚Üí        h.make_node("Constant", [], ["g_starts"],
   362‚Üí   362‚Üí                     value=h.make_tensor("g_starts_v", TensorProto.INT64, [4], [0, 0, 0, 0])),
   363‚Üí   363‚Üí        h.make_node("Constant", [], ["g_ends"],
   364‚Üí   364‚Üí                     value=h.make_tensor("g_ends_v", TensorProto.INT64, [4], [1, 1, kk, kk])),
   365‚Üí   365‚Üí        h.make_node("Constant", [], ["g_axes"],
   366‚Üí   366‚Üí                     value=h.make_tensor("g_axes_v", TensorProto.INT64, [4], [0, 1, 2, 3])),
   367‚Üí   367‚Üí        h.make_node("Slice", ["T1", "g_starts", "g_ends", "g_axes"], ["T1_c0"]),
   368‚Üí   368‚Üí        # gate = 1 - T1_c0 (broadcasts: (1,1,1,1) - (1,1,k*k,k*k) ‚Üí (1,1,k*k,k*k))
   369‚Üí   369‚Üí        h.make_node("Constant", [], ["one"],
   370‚Üí   370‚Üí                     value=h.make_tensor("one_v", TensorProto.FLOAT, [1, 1, 1, 1], [1.0])),

   372‚Üí   372‚Üí        # O_small = T2 * gate (broadcasts channel)
   373‚Üí   373‚Üí        h.make_node("Mul", ["T2", "gate"], ["O_small"]),

   377‚Üí   377‚Üí                                          [0, 0, 0, 0, 0, 0, pad_h, pad_w])),
   378‚Üí   378‚Üí        h.make_node("Constant", [], ["pad_val"],
   379‚Üí   379‚Üí                     value=h.make_tensor("pad_val_v", TensorProto.FLOAT, [], [0.0])),
   380‚Üí   380‚Üí        h.make_node("Pad", ["O_small", "pad_c", "pad_val"], [OUTPUT_NAME], mode="constant"),
   381‚Üí   381‚Üí    ]
   382‚Üí   382‚Üí    return _make_simple_model(nodes, name=f"kronecker_{k}x{k}")
   383‚Üí   383‚Üí
   384‚Üí   384‚Üí
   385‚Üí   385‚Üí# ---------------------------------------------------------------------------
   386‚Üí   386‚Üí# ConcatRepeatSolver ‚Äî extend input by repeating first K rows (or cols)
   387‚Üí   387‚Üí# ---------------------------------------------------------------------------
   388‚Üí   388‚Üí
   389‚Üí   389‚Üí
   390‚Üí   390‚Üíclass ConcatRepeatSolver(Solver):
   391‚Üí   391‚Üí    """Output = input + first K rows of input appended at the bottom (or similar).
   392‚Üí   392‚Üí
   393‚Üí   393‚Üí    Matches task 017c7c7b (6x3 -> 9x3, output = input + first 3 rows).
   394‚Üí   394‚Üí    """
   395‚Üí   395‚Üí    name = "concat_repeat"
   396‚Üí   396‚Üí
   397‚Üí   397‚Üí    def attempt(self, task: dict) -> Optional[onnx.ModelProto]:
   398‚Üí   398‚Üí        pairs = get_pairs(task)
   399‚Üí   399‚Üí        # Determine direction (rows or cols) and K
   400‚Üí   400‚Üí        direction = None
   401‚Üí   401‚Üí        K = None
   402‚Üí   402‚Üí        for inp, out in pairs:
   403‚Üí   403‚Üí            H_in, W_in = inp.shape
   404‚Üí   404‚Üí            H_out, W_out = out.shape
   405‚Üí   405‚Üí            # Check rows extension
   406‚Üí   406‚Üí            if W_in == W_out and H_out > H_in:
   407‚Üí   407‚Üí                # Check if out[:H_in] == inp and out[H_in:] == inp[:K] for some K
   408‚Üí   408‚Üí                if not np.array_equal(out[:H_in], inp):
   409‚Üí   409‚Üí                    return None
   410‚Üí   410‚Üí                K_candidate = H_out - H_in
   411‚Üí   411‚Üí                if not np.array_equal(out[H_in:], inp[:K_candidate]):
   412‚Üí   412‚Üí                    return None
   413‚Üí   413‚Üí                if direction is None:
   414‚Üí   414‚Üí                    direction = "rows"
   415‚Üí   415‚Üí                    K = K_candidate
   416‚Üí   416‚Üí                elif direction != "rows" or K != K_candidate:
   417‚Üí   417‚Üí                    return None
   418‚Üí   418‚Üí            elif H_in == H_out and W_out > W_in:
   419‚Üí   419‚Üí                if not np.array_equal(out[:, :W_in], inp):
   420‚Üí   420‚Üí                    return None
   421‚Üí   421‚Üí                K_candidate = W_out - W_in
   422‚Üí   422‚Üí                if not np.array_equal(out[:, W_in:], inp[:, :K_candidate]):
   423‚Üí   423‚Üí                    return None
   424‚Üí   424‚Üí                if direction is None:
   425‚Üí   425‚Üí                    direction = "cols"
   426‚Üí   426‚Üí                    K = K_candidate
   427‚Üí   427‚Üí                elif direction != "cols" or K != K_candidate:
   428‚Üí   428‚Üí                    return None
   429‚Üí   429‚Üí            else:
   430‚Üí   430‚Üí                return None
   431‚Üí   431‚Üí        if direction is None:
   432‚Üí   432‚Üí            return None
   433‚Üí   433‚Üí        return _concat_repeat_model(direction, K)

   436‚Üí   436‚Üídef _concat_repeat_model(direction: str, K: int) -> onnx.ModelProto:
   437‚Üí   437‚Üí    """Build a network that outputs [input, input[:K]] (rows) or [input | input[:,:K]] (cols)
   438‚Üí   438‚Üí    in the top-left of the 30x30 frame.
   439‚Üí   439‚Üí    """
   440‚Üí   440‚Üí    if direction == "rows":
   441‚Üí   441‚Üí        # Slice input[:K] (top K rows) ‚Äî slice on axis 2
   442‚Üí   442‚Üí        nodes = [
   443‚Üí   443‚Üí            h.make_node("Constant", [], ["starts"],
   444‚Üí   444‚Üí                         value=h.make_tensor("starts_v", TensorProto.INT64, [4], [0, 0, 0, 0])),
   445‚Üí   445‚Üí            h.make_node("Constant", [], ["ends"],
   446‚Üí   446‚Üí                         value=h.make_tensor("ends_v", TensorProto.INT64, [4], [1, NUM_COLORS, K, MAX_GRID])),
   447‚Üí   447‚Üí            h.make_node("Constant", [], ["axes"],
   448‚Üí   448‚Üí                         value=h.make_tensor("axes_v", TensorProto.INT64, [4], [0, 1, 2, 3])),
   449‚Üí   449‚Üí            h.make_node("Slice", [INPUT_NAME, "starts", "ends", "axes"], ["top_K"]),
   450‚Üí   450‚Üí            # Concat along axis 2 (rows): input + top_K
   451‚Üí   451‚Üí            h.make_node("Concat", [INPUT_NAME, "top_K"], [OUTPUT_NAME], axis=2),
   452‚Üí   452‚Üí        ]
   453‚Üí   453‚Üí    else:  # cols
   454‚Üí   454‚Üí        nodes = [
   455‚Üí   455‚Üí            h.make_node("Constant", [], ["starts"],
   456‚Üí   456‚Üí                         value=h.make_tensor("starts_v", TensorProto.INT64, [4], [0, 0, 0, 0])),
   457‚Üí   457‚Üí            h.make_node("Constant", [], ["ends"],
   458‚Üí   458‚Üí                         value=h.make_tensor("ends_v", TensorProto.INT64, [4], [1, NUM_COLORS, MAX_GRID, K])),
   459‚Üí   459‚Üí            h.make_node("Constant", [], ["axes"],
   460‚Üí   460‚Üí                         value=h.make_tensor("axes_v", TensorProto.INT64, [4], [0, 1, 2, 3])),
   461‚Üí   461‚Üí            h.make_node("Slice", [INPUT_NAME, "starts", "ends", "axes"], ["left_K"]),
   462‚Üí   462‚Üí            h.make_node("Concat", [INPUT_NAME, "left_K"], [OUTPUT_NAME], axis=3),
   463‚Üí   463‚Üí        ]
   464‚Üí   464‚Üí    return _make_simple_model(nodes, name=f"concat_repeat_{direction}_{K}")

   576‚Üí   576‚Üídef _slice_colormap_model(axis: str, idx: int, side: str, mapping: dict[int, int]) -> onnx.ModelProto:
   577‚Üí   577‚Üí    """Build a network that slices the input on one side of a separator line
   578‚Üí   578‚Üí    and applies a color map to the result.
   579‚Üí   579‚Üí
   580‚Üí   580‚Üí    axis = "col" or "row"; idx = separator index; side = "left"/"right"/"top"/"bottom"
   581‚Üí   581‚Üí    """
   582‚Üí   582‚Üí    # Step 1: slice
   583‚Üí   583‚Üí    if axis == "col":
   584‚Üí   584‚Üí        if side == "left":
