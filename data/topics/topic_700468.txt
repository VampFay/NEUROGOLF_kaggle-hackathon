Possible issue with Task 014 — solution verified on all public data, yet unscoreable

Hi organizers and fellow competitors,

We'd like to flag what looks like a possible issue with Task 014 (ARC-AGI training task 0b148d64).

The task is very simple. The input grid is split into several rectangular regions by bands of background (0) cells. All regions but one share the same color; exactly one region has a different color. The output is simply that odd-colored region, cropped out as-is. There is no ambiguity — every training example agrees on this single rule.

Our solution is verified on 100% of the public data. When we submit this solution, the leaderboard score does not change. As a control, we also submitted a deliberately wrong (identity) network for Task 014 — the score still did not change.

Our concern Given how simple and unambiguous this task is, and that our solution provably matches every public example, we suspect there may be an error in Task 014 — most plausibly a mislabeled example in the private benchmark.

Could the organizers please double-check Task 014? Thanks!

This is my code, and I would appreciate it if you could help review it. Additionally, I am mainly working on the train+test+arc-gen tests provided by this competition.

from pathlib import Path

import numpy as np
import torch

ROOT = Path('/workspace/2026-NeuroGolf-Championship')
OUT = ROOT / 'results/task014_rule_solver/task014.onnx'
BIG = 99.0


class Solver(torch.nn.Module):
    def forward(self, x):                       # x: [1,10,30,30] one-hot float
        idx = torch.arange(30, dtype=torch.float32)
        r_ax = idx.view(1, 1, 30, 1)
        c_ax = idx.view(1, 1, 1, 30)

        fg = x[:, 1:10]                          # [1,9,30,30] 非零色通道
        nz = fg.sum(1, keepdim=True)             # [1,1,30,30] 该格是否非零
        row_nz = nz.amax(3, keepdim=True)        # [1,1,30,1] 该行是否含非零
        col_nz = nz.amax(2, keepdim=True)        # [1,1,1,30]
        zr = 1.0 - row_nz                        # [1,1,30,1] 该行是「全0行/padding」
        zc = 1.0 - col_nz                        # [1,1,1,30]

        # 每个非零色的像素行/列范围
        row_has = fg.amax(3, keepdim=True)       # [1,9,30,1]
        col_has = fg.amax(2, keepdim=True)       # [1,9,1,30]
        minr = (r_ax * row_has + BIG * (1 - row_has)).amin(2, keepdim=True)  # [1,9,1,1]
        maxr = (r_ax * row_has).amax(2, keepdim=True)
        minc = (c_ax * col_has + BIG * (1 - col_has)).amin(3, keepdim=True)
        maxc = (c_ax * col_has).amax(3, keepdim=True)
        exists = fg.amax((2, 3), keepdim=True)   # [1,9,1,1]

        # 独特色：行/列范围内不跨任何全0带
        in_r = (r_ax >= minr).float() * (r_ax <= maxr).float()   # [1,9,30,1]
        in_c = (c_ax >= minc).float() * (c_ax <= maxc).float()   # [1,9,1,30]
        row_gap = (zr * in_r).sum(2, keepdim=True)               # [1,9,1,1]
        col_gap = (zc * in_c).sum(3, keepdim=True)
        is_u = (row_gap < 0.5).float() * (col_gap < 0.5).float() \
            * (exists > 0.5).float()                             # [1,9,1,1] 恰一个为1

        # 独特色的像素范围
        minr_u = (is_u * minr).sum(1, keepdim=True)              # [1,1,1,1]
        maxr_u = (is_u * maxr).sum(1, keepdim=True)
        minc_u = (is_u * minc).sum(1, keepdim=True)
        maxc_u = (is_u * maxc).sum(1, keepdim=True)

        # 把独特色范围扩展成完整 part 矩形（扩到最近的全0行/列带）
        below = (r_ax < minr_u).float()
        r0 = ((r_ax + 1.0) * zr * below).amax(2, keepdim=True)   # [1,1,1,1]
        above = (r_ax > maxr_u).float()
        a_r = zr * above
        r1 = ((r_ax - 1.0) * a_r + (1.0 - a_r) * BIG).amin(2, keepdim=True)
        r1 = torch.clamp(r1, max=29.0)

        below_c = (c_ax < minc_u).float()
        c0 = ((c_ax + 1.0) * zc * below_c).amax(3, keepdim=True)
        above_c = (c_ax > maxc_u).float()
        a_c = zc * above_c
        c1 = ((c_ax - 1.0) * a_c + (1.0 - a_c) * BIG).amin(3, keepdim=True)
        c1 = torch.clamp(c1, max=29.0)

        r0s, c0s = r0.reshape(()), c0.reshape(())
        h = (r1 - r0 + 1.0).reshape(())
        w = (c1 - c0 + 1.0).reshape(())

        # 平移置换矩阵：把 part 矩形搬到左上角
        k_grid = idx.view(30, 1)
        j_grid = idx.view(1, 30)
        pr = (torch.abs(j_grid - (k_grid + r0s)) < 0.5).float()  # [30,30]
        pc = (torch.abs(j_grid - (k_grid + c0s)) < 0.5).float()  # [30,30]
        mask = (idx.view(30, 1) < h).float() * (idx.view(1, 30) < w).float()

        xs = x.reshape(10, 30, 30)
        shifted = torch.matmul(pr, xs)              # 上移 r0 行
        shifted = torch.matmul(shifted, pc.t())     # 左移 c0 列
        out = shifted * mask                        # 清掉矩形外
        return out.reshape(1, 10, 30, 30)


def encode(grid):
    x = np.zeros((1, 10, 30, 30), np.float32)
    for r, row in enumerate(grid):
        for c, v in enumerate(row):
            x[0, v, r, c] = 1.0
    return x


def build_and_verify():
    import json
    import onnx
    import onnxruntime as ort

    OUT.parent.mkdir(parents=True, exist_ok=True)
    model = Solver().eval()
    with torch.no_grad():
        torch.onnx.export(
            model, torch.zeros(1, 10, 30, 30), str(OUT),
            input_names=['input'], output_names=['output'],
            opset_version=13, do_constant_folding=True,
        )
    m = onnx.load(str(OUT))
    banned = {'Loop', 'Scan', 'NonZero', 'Unique', 'Script', 'Function'}
    used = {n.op_type for n in m.graph.node}
    assert not (used & banned), f'banned ops: {used & banned}'
    print(f'built {OUT} | nodes={len(m.graph.node)} | size={OUT.stat().st_size} bytes')

    opts = ort.SessionOptions()
    opts.graph_optimization_level = ort.GraphOptimizationLevel.ORT_DISABLE_ALL
    sess = ort.InferenceSession(str(OUT), opts, providers=['CPUExecutionProvider'])
    d = json.load(open(ROOT / 'data/task014.json'))
    ok = total = 0
    bad = []
    for split in ['train', 'test', 'arc-gen']:
        for i, ex in enumerate(d.get(split, [])):
            if 'output' not in ex:
                continue
            total += 1
            pred = (sess.run(['output'], {'input': encode(ex['input'])})[0] > 0)
            if np.array_equal(pred.astype(np.float32), encode(ex['output'])):
                ok += 1
            else:
                bad.append((split, i))
    print(f'onnx validate: {ok}/{total}')
    if bad:
        print('first bad:', bad[:8])


if __name__ == '__main__':
    build_and_verify()
